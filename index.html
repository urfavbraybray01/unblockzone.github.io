const reviews = [ { stars: '⭐⭐⭐⭐⭐', name: 'John Doe', description: 'Amazing service and fast speeds!' }, { stars: '⭐⭐⭐⭐⭐', name: 'Jane Smith', description: 'Highly recommend for unblocking content.' }, { stars: '⭐⭐⭐⭐', name: 'Mark Lee', description: 'Good performance, but a bit slow at times.' }, { stars: '⭐⭐⭐⭐⭐', name: 'Sara Black', description: 'The best experience for streaming.' }, { stars: '⭐⭐⭐⭐⭐', name: 'Chris Brown', description: 'Very reliable and great customer support.' }, { stars: '⭐⭐⭐⭐', name: 'Lisa Green', description: 'Decent, but sometimes disconnects.' }, { stars: '⭐⭐⭐⭐⭐', name: 'David White', description: 'Incredible speed and uptime!' }, { stars: '⭐⭐⭐⭐', name: 'Nina Gray', description: 'Pretty solid, but can be a bit laggy.' }, ]; let currentReviewIndex = 0; const reviewsPerPage = 4; // Load reviews based on current index function loadReviews() { const reviewContainer = document.getElementById('reviews'); // Add fade-out class before updating content reviewContainer.classList.add('fade-out'); // Wait for fade-out animation to complete before updating content setTimeout(() => { reviewContainer.innerHTML = ''; const visibleReviews = reviews.slice(currentReviewIndex, currentReviewIndex + reviewsPerPage); visibleReviews.forEach(review => { const reviewBox = document.createElement('div'); reviewBox.classList.add('review-box'); reviewBox.innerHTML = `
${review.stars}
${review.name}
${review.description}
`; reviewContainer.appendChild(reviewBox); }); // Remove the fade-out class and add fade-in reviewContainer.classList.remove('fade-out'); reviewContainer.classList.add('fade-in'); // Control arrow visibility if (currentReviewIndex + reviewsPerPage >= reviews.length) { document.getElementById('nextReviews').style.display = 'none'; document.getElementById('prevReviews').style.display = 'block'; } else if (currentReviewIndex === 0) { document.getElementById('prevReviews').style.display = 'none'; document.getElementById('nextReviews').style.display = 'block'; } else { document.getElementById('nextReviews').style.display = 'block'; document.getElementById('prevReviews').style.display = 'block'; } }, 700); // Match the fade-out duration (0.7s) } // Load next set of reviews function loadNextReviews() { if (currentReviewIndex + reviewsPerPage < reviews.length) { currentReviewIndex += reviewsPerPage; loadReviews(); } } // Load previous set of reviews function loadPrevReviews() { if (currentReviewIndex - reviewsPerPage >= 0) { currentReviewIndex -= reviewsPerPage; loadReviews(); } } // Event listeners for arrows document.getElementById('nextReviews').addEventListener('click', loadNextReviews); document.getElementById('prevReviews').addEventListener('click', loadPrevReviews); // Initial load loadReviews();
